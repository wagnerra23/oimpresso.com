public function updateLedgerTotal(int $id, float $novoTotal): void
{
    // findOrFail(id-do-request) sem where('business_id', ...) numa tabela tenant-owned.
    $ledger = Ledger::findOrFail($id);
    $ledger->update(['final_total' => $novoTotal]);
}
