/**
 * @return int|null Quantidade em estoque; null quando o produto não existe.
 */
public function estoqueDe(int $productId): ?int
{
    $product = Product::where('business_id', session('user.business_id'))
        ->where('id', $productId)
        ->first();

    if ($product === null) {
        return null;
    }
    return (int) $product->on_hand;
}
